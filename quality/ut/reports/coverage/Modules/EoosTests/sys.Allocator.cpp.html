#include "sys.Call.hpp"

namespace eoos
{
namespace sys
{
 
void* Allocator::allocate(size_t size)
{
    return sys::Call::get().getHeap().allocate(size, NULLPTR);
}

void Allocator::free(void* ptr)
{
    return sys::Call::get().getHeap().free(ptr);    
}
